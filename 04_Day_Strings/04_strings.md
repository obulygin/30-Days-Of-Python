# –î–µ–Ω—å 4

–¢–µ–∫—Å—Ç –≤ Python —è–≤–ª—è–µ—Ç—Å—è —Ç–∏–ø–æ–º –¥–∞–Ω–Ω—ã—Ö "—Å—Ç—Ä–æ–∫–∞"(string). –õ—é–±–æ–π —Ç–∏–ø –¥–∞–Ω–Ω—ã—Ö, –∑–∞–ø–∏—Å–∞–Ω–Ω—ã–π –≤ –≤–∏–¥–µ —Ç–µ–∫—Å—Ç–∞, —è–≤–ª—è–µ—Ç—Å—è —Å—Ç—Ä–æ–∫–æ–π. –õ—é–±—ã–µ –¥–∞–Ω–Ω—ã–µ, –∑–∞–∫–ª—é—á–µ–Ω–Ω—ã–µ –≤ –æ–¥–∏–Ω–∞—Ä–Ω—ã–µ, –¥–≤–æ–π–Ω—ã–µ –∏–ª–∏ —Ç—Ä–æ–π–Ω—ã–µ –∫–∞–≤—ã—á–∫–∏, —è–≤–ª—è—é—Ç—Å—è —Å—Ç—Ä–æ–∫–∞–º–∏. –í Python —Å—É—â–µ—Å—Ç–≤—É—é—Ç —Ä–∞–∑–ª–∏—á–Ω—ã–µ –º–µ—Ç–æ–¥—ã —Å—Ç—Ä–æ–∫ –∏ –≤—Å—Ç—Ä–æ–µ–Ω–Ω—ã–µ —Ñ—É–Ω–∫—Ü–∏–∏ –¥–ª—è —Ä–∞–±–æ—Ç—ã —Å –¥–∞–Ω–Ω—ã–º–∏ —Ç–∏–ø–∞ "—Å—Ç—Ä–æ–∫–∞". –ß—Ç–æ–±—ã –ø—Ä–æ–≤–µ—Ä–∏—Ç—å –¥–ª–∏–Ω—É —Å—Ç—Ä–æ–∫–∏, –∏—Å–ø–æ–ª—å–∑—É–π—Ç–µ —Ñ—É–Ω–∫—Ü–∏—é len().

### –°–æ–∑–¥–∞–Ω–∏–µ —Å—Ç—Ä–æ–∫

–ü—Ä–∏–º–µ—Ä:

### –ö–æ–Ω–∫–∞—Ç–µ–Ω–∞—Ü–∏—è —Å—Ç—Ä–æ–∫

–ú—ã –º–æ–∂–µ–º –æ–±—ä–µ–¥–∏–Ω—è—Ç—å —Å—Ç—Ä–æ–∫–∏. –û–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ —Å—Ç—Ä–æ–∫ –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è –∫–æ–Ω–∫–∞—Ç–µ–Ω–∞—Ü–∏–µ–π.

### –≠–∫—Ä–∞–Ω–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏

–î–∞–≤–∞–π—Ç–µ –≤–∑–≥–ª—è–Ω–µ–º –Ω–∞ –Ω–∞–∏–±–æ–ª–µ–µ —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–Ω—ã–µ —ç–∫—Ä–∞–Ω–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ —Å–∏–º–≤–æ–ª—ã:

- \\\\: –û–±—Ä–∞—Ç–Ω–∞—è –∫–æ—Å–∞—è —á–µ—Ä—Ç–∞
- \\': –û–¥–∏–Ω–∞—Ä–Ω—ã–µ –∫–∞–≤—ã—á–∫–∏ (')
- \\": –î–≤–æ–π–Ω—ã–µ –∫–∞–≤—ã—á–∫–∏ (")

### –§–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ —Å—Ç—Ä–æ–∫

#### –°—Ç–∞—Ä—ã–π —Å—Ç–∏–ª—å —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è —Å—Ç—Ä–æ–∫ (–æ–ø–µ—Ä–∞—Ç–æ—Ä %)

–í Python —Å—É—â–µ—Å—Ç–≤—É–µ—Ç –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Å–ø–æ—Å–æ–±–æ–≤ —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è —Å—Ç—Ä–æ–∫. –°–µ–π—á–∞—Å –º—ã —Ä–∞—Å—Å–º–æ—Ç—Ä–∏–º –Ω–µ–∫–æ—Ç–æ—Ä—ã–µ –∏–∑ –Ω–∏—Ö. –û–ø–µ—Ä–∞—Ç–æ—Ä "%" –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –¥–ª—è —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è –Ω–∞–±–æ—Ä–∞ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö –æ–±—ä–µ–¥–∏–Ω–µ–Ω–Ω—ã—Ö –≤ –∫–æ—Ä—Ç–µ–∂ (tuple), –≤–º–µ—Å—Ç–µ —Å —Ñ–æ—Ä–º–∞—Ç–∏—Ä—É—é—â–µ–π —Å—Ç—Ä–æ–∫–æ–π, –∫–æ—Ç–æ—Ä–∞—è —Å–æ–¥–µ—Ä–∂–∏—Ç –æ–±—ã—á–Ω—ã–π —Ç–µ–∫—Å—Ç —Å–æ "—Å–ø–µ—Ü–∏—Ñ–∏–∫–∞—Ç–æ—Ä–∞–º–∏ –∞—Ä–≥—É–º–µ–Ω—Ç–æ–≤" - —Å–ø–µ—Ü–∏–∞–ª—å–Ω—ã–º–∏ —Å–∏–º–≤–æ–ª–∞–º–∏, —Ç–∞–∫–∏–º–∏ –∫–∞–∫ "%s", "%d", "%f", "%.2f".

- %s - –°—Ç—Ä–æ–∫–∞ (–∏–ª–∏ –ª—é–±–æ–π –æ–±—ä–µ–∫—Ç —Å–æ —Å—Ç—Ä–æ–∫–æ–≤—ã–º –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ–º, –Ω–∞–ø—Ä–∏–º–µ—Ä, —á–∏—Å–ª–∞)
- %d - –¶–µ–ª—ã–µ —á–∏—Å–ª–∞
- %f - –ß–∏—Å–ª–∞ —Å –ø–ª–∞–≤–∞—é—â–µ–π —Ç–æ—á–∫–æ–π
- ""%.2f" - –ß–∏—Å–ª–∞ —Å –ø–ª–∞–≤–∞—é—â–µ–π —Ç–æ—á–∫–æ–π —Å —Ç–æ—á–Ω–æ—Å—Ç—å—é –¥–æ —É–∫–∞–∑–∞–Ω–Ω–æ–≥–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –∑–Ω–∞–∫–æ–≤

```py
# –¢–æ–ª—å–∫–æ —Å—Ç—Ä–æ–∫–∏
first_name = '–ü–∏—Ç–µ—Ä'
last_name = '–ü–∞—Ä–∫–µ—Ä'
language = 'Python'
formated_string = '–ú–µ–Ω—è –∑–æ–≤—É—Ç %s %s. –Ø —É—á—É %s.' %(first_name, last_name, language)
print(formated_string) # –ú–µ–Ω—è –∑–æ–≤—É—Ç –ü–∏—Ç–µ—Ä –ü–∞—Ä–∫–µ—Ä. –Ø —É—á—É Python.

# –°—Ç—Ä–æ–∫–∏ –∏ —á–∏—Å–ª–∞
radius = 10
pi = 3.14
area = pi * radius ** 2
formated_string = '–ü–ª–æ—â–∞–¥—å –∫—Ä—É–≥–∞ —Å —Ä–∞–¥–∏—É—Å–æ–º %d —Ä–∞–≤–Ω–∞ %.2f.' %(radius, area) # 2 —É–∫–∞–∑—ã–≤–∞–µ—Ç –Ω–∞ 2 –∑–Ω–∞—á–∞—â–∏–µ —Ü–∏—Ñ—Ä—ã –ø–æ—Å–ª–µ —Ç–æ—á–∫–∏

python_libraries = ['Django', 'Flask', 'NumPy', 'Matplotlib','Pandas']
formated_string = '–ï—Å—Ç—å —Ä–∞–∑–Ω—ã–µ –±–∏–±–ª–∏–æ—Ç–µ–∫–∏ –¥–ª—è Python. –í–æ—Ç –Ω–µ–∫–æ—Ç–æ—Ä—ã–µ –∏–∑ –Ω–∏—Ö: %s' % (python_libraries)
print(formated_string) # "–ï—Å—Ç—å —Ä–∞–∑–Ω—ã–µ –±–∏–±–ª–∏–æ—Ç–µ–∫–∏ python. –í–æ—Ç –Ω–µ–∫–æ—Ç–æ—Ä—ã–µ –∏–∑ –Ω–∏—Ö: ['Django', 'Flask', 'NumPy', 'Matplotlib','Pandas']"
```

#### –ù–æ–≤—ã–π —Å—Ç–∏–ª—å —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è —Å—Ç—Ä–æ–∫ (str.format)

–≠—Ç–æ—Ç —Å–ø–æ—Å–æ–± —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è —Å—Ç—Ä–æ–∫ —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º –º–µ—Ç–æ–¥–∞ format() –±—ã–ª –≤–≤–µ–¥–µ–Ω –≤ Python –≤–µ—Ä—Å–∏–∏ 3. –û–Ω –ø—Ä–µ–¥–æ—Å—Ç–∞–≤–ª—è–µ—Ç –±–æ–ª–µ–µ –≥–∏–±–∫–∏–π –∏ —É–Ω–∏–≤–µ—Ä—Å–∞–ª—å–Ω—ã–π –ø–æ–¥—Ö–æ–¥ –∫ —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—é —Å—Ç—Ä–æ–∫ –ø–æ —Å—Ä–∞–≤–Ω–µ–Ω–∏—é —Å –ø—Ä–µ–¥—ã–¥—É—â–∏–º (–æ–ø–µ—Ä–∞—Ç–æ—Ä %).

```py
first_name = '–ü–∏—Ç–µ—Ä'
last_name = '–ü–∞—Ä–∫–µ—Ä'
language = 'Python'
formated_string = '–ú–µ–Ω—è –∑–æ–≤—É—Ç {} {}. –Ø —É—á–∏—Ç–µ–ª—å —è–∑—ã–∫–∞ –ø—Ä–æ–≥—Ä–∞–º–º–∏—Ä–æ–≤–∞–Ω–∏—è {}.'.format(first_name, last_name, language)
print(formated_string) # –ú–µ–Ω—è –∑–æ–≤—É—Ç –ü–∏—Ç–µ—Ä –ü–∞—Ä–∫–µ—Ä. –Ø —É—á–∏—Ç–µ–ª—å —è–∑—ã–∫–∞ –ø—Ä–æ–≥—Ä–∞–º–º–∏—Ä–æ–≤–∞–Ω–∏—è Python.

a = 4
b = 3

print('{} + {} = {}'.format(a, b, a + b))
print('{} - {} = {}'.format(a, b, a - b))
print('{} * {} = {}'.format(a, b, a * b))
print('{} / {} = {:.2f}'.format(a, b, a / b)) # –æ–≥—Ä–∞–Ω–∏—á–∏–≤–∞–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ü–∏—Ñ—Ä –ø–æ—Å–ª–µ –∑–∞–ø—è—Ç–æ–π –¥–æ –¥–≤—É—Ö.
print('{} % {} = {}'.format(a, b, a % b))
print('{} // {} = {}'.format(a, b, a // b))
print('{} ** {} = {}'.format(a, b, a ** b))

# –í—ã–≤–æ–¥–∏—Ç
4 + 3 = 7
4 - 3 = 1
4 * 3 = 12
4 / 3 = 1.33
4 % 3 = 1
4 // 3 = 1
4 ** 3 = 64

# –°—Ç—Ä–æ–∫–∏ –∏ —á–∏—Å–ª–∞
radius = 10
pi = 3.14
area = pi * radius ** 2
formated_string = '–ü–ª–æ—â–∞–¥—å –∫—Ä—É–≥–∞ —Ä–∞–¥–∏—É—Å–æ–º {} —Å–æ—Å—Ç–∞–≤–ª—è–µ—Ç {:.2f}.'.format(radius, area) # 2 —á–∏—Å–ª–∞ –ø–æ—Å–ª–µ –∑–∞–ø—è—Ç–æ–π
print(formated_string)

```

#### –°–∞–º—ã–π –Ω–æ–≤—ã–π —Å–ø–æ—Å–æ–± —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è —Å—Ç—Ä–æ–∫ - f-—Å—Ç—Ä–æ–∫–∏ (Python 3.6+)

–ï—â–µ –æ–¥–∏–Ω —Å–ø–æ—Å–æ–± —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è —Å—Ç—Ä–æ–∫ - —ç—Ç–æ f-—Å—Ç—Ä–æ–∫. –ï—Å–ª–∏ –ø–µ—Ä–µ–¥ —Å—Ç—Ä–æ–∫–æ–π –ø–æ—Å—Ç–∞–≤–∏—Ç—å —Å–∏–º–≤–æ–ª "f", —Ç–æ —ç—Ç–æ –¥–∞—Å—Ç –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç—å –≤—Å—Ç–∞–≤–ª—è—Ç—å –∑–Ω–∞—á–µ–Ω–∏—è (–∏ –¥–∞–∂–µ –æ–¥–Ω–æ—Å—Ç—Ä–æ—á–Ω—ã–µ –æ–ø–µ—Ä–∞—Ü–∏–∏) –Ω–∞ –Ω—É–∂–Ω—ã–µ –ø–æ–∑–∏—Ü–∏–∏ –≤–Ω—É—Ç—Ä–∏ —Å—Ç—Ä–æ–∫. 

```py
a = 4
b = 3
print(f'{a} + {b} = {a + b}')
print(f'{a} - {b} = {a - b}')
print(f'{a} * {b} = {a * b}')
print(f'{a} / {b} = {a / b:.2f}')
print(f'{a} % {b} = {a % b}')
print(f'{a} // {b} = {a // b}')
print(f'{a} ** {b} = {a ** b}')
```

### –°—Ç—Ä–æ–∫–∏ –≤ Python –∫–∞–∫ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ —Å–∏–º–≤–æ–ª–æ–≤

–°—Ç—Ä–æ–∫–∏ –≤ Python —è–≤–ª—è—é—Ç—Å—è –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—è–º–∏ —Å–∏–º–≤–æ–ª–æ–≤ –∏ –æ–±–ª–∞–¥–∞—é—Ç –±–∞–∑–æ–≤—ã–º–∏ –º–µ—Ç–æ–¥–∞–º–∏, —Å—Ö–æ–∂–∏–º–∏ —Å –¥—Ä—É–≥–∏–º–∏ —É–ø–æ—Ä—è–¥–æ—á–µ–Ω–Ω—ã–º–∏ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—è–º–∏ –æ–±—ä–µ–∫—Ç–æ–≤ –≤ Python, —Ç–∞–∫–∏–º–∏ –∫–∞–∫ —Å–ø–∏—Å–∫–∏ –∏ –∫–æ—Ä—Ç–µ–∂–∏. –°–∞–º—ã–π –ø—Ä–æ—Å—Ç–æ–π —Å–ø–æ—Å–æ–± –∏–∑–≤–ª–µ—á–µ–Ω–∏—è –æ—Ç–¥–µ–ª—å–Ω—ã—Ö —Å–∏–º–≤–æ–ª–æ–≤ –∏–∑ —Å—Ç—Ä–æ–∫ (–∞ —Ç–∞–∫–∂–µ –æ—Ç–¥–µ–ª—å–Ω—ã—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –∏–∑ –ª—é–±–æ–π –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏) - —ç—Ç–æ —Ä–∞—Å–ø–∞–∫–æ–≤–∫–∞ –∏—Ö –≤ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–µ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ.

#### –†–∞—Å–ø–∞–∫–æ–≤–∫–∞ —Å—Ç—Ä–æ–∫

```py
language = 'Python'
a, b, c, d, e, f = language # —Ä–∞—Å–ø–∞–∫–æ–≤–∫–∞ —Å—Ç—Ä–æ–∫–∏ –≤ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å –ø–µ—Ä–µ–º–µ–Ω–Ω—ã–µ
print(a) # P
print(b) # y
print(c) # t
print(d) # h
print(e) # o
print(f) # n
```


#### –î–æ—Å—Ç—É–ø –∫ —Å–∏–º–≤–æ–ª–∞–º –≤ —Å—Ç—Ä–æ–∫–∞—Ö –ø–æ –∏–Ω–¥–µ–∫—Å—É

–í –ø—Ä–æ–≥—Ä–∞–º–º–∏—Ä–æ–≤–∞–Ω–∏–∏ –Ω—É–º–µ—Ä–∞—Ü–∏—è –Ω–∞—á–∏–Ω–∞–µ—Ç—Å—è —Å –Ω—É–ª—è. –ü–æ—ç—Ç–æ–º—É –ø–µ—Ä–≤–∞—è –±—É–∫–≤–∞ —Å—Ç—Ä–æ–∫–∏ –Ω–∞—Ö–æ–¥–∏—Ç—Å—è –ø–æ –∏–Ω–¥–µ–∫—Å—É 0, –∞ –ø–æ—Å–ª–µ–¥–Ω—è—è –±—É–∫–≤–∞ - –ø–æ –∏–Ω–¥–µ–∫—Å—É –¥–ª–∏–Ω—ã —Å—Ç—Ä–æ–∫–∏ –º–∏–Ω—É—Å –æ–¥–∏–Ω.

![String index](../images/string_index.png)

```py
language = 'Python'
first_letter = language[0]
print(first_letter)  # P
second_letter = language[1]
print(second_letter)  # y
last_index = len(language) - 1
last_letter = language[last_index]
print(last_letter)  # n
```

–ï—Å–ª–∏ –º—ã —Ö–æ—Ç–∏–º –Ω–∞—á–∞—Ç—å —Å –∫–æ–Ω—Ü–∞ —Å—Ç—Ä–æ–∫–∏, –º—ã –º–æ–∂–µ–º –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–π –∏–Ω–¥–µ–∫—Å. -1 –æ–±–æ–∑–Ω–∞—á–∞–µ—Ç –∏–Ω–¥–µ–∫—Å –ø–æ—Å–ª–µ–¥–Ω–µ–≥–æ —Å–∏–º–≤–æ–ª–∞, -2 –∏–Ω–¥–µ–∫—Å –≤—Ç–æ—Ä–æ–≥–æ —Å–∏–º–≤–æ–ª–∞ —Å –∫–æ–Ω—Ü–∞ –∏ —Ç.–¥.

```py
language = 'Python'
last_letter = language[-1]
print(last_letter)  # n
second_last = language[-2]
print(second_last)  # o
```

#### –°—Ä–µ–∑—ã —Å—Ç—Ä–æ–∫ –≤ Python

–í Python –º—ã –º–æ–∂–µ–º –±—Ä–∞—Ç—å —Å—Ä–µ–∑—ã –æ—Ç —Å—Ç—Ä–æ–∫ –ø—É—Ç–µ–º –æ–±—Ä–∞—â–µ–Ω–∏—è –∫ –∏–Ω–¥–µ–∫—Å–∞–º —ç—Ç–æ–π —Å—Ç—Ä–æ–∫–∏.

```py
language = 'Python'
first_three = language[0:3]  # –Ω–∞—á–∏–Ω–∞–µ–º —Å –Ω—É–ª–µ–≤–æ–≥–æ –∏–Ω–¥–µ–∫—Å–∞ –≤–∫–ª—é—á–∏—Ç–µ–ª—å–Ω–æ –∏ –∏–¥–µ–º –¥–æ 3 –Ω–µ–≤–∫–ª—é—á–∏—Ç–µ–ª—å–Ω–æ. 
print(first_three) # Pyt
last_three = language[3:6]
print(last_three)  # hon
# –î—Ä—É–≥–æ–π —Å–ø–æ—Å–æ–±
last_three = language[-3:]
print(last_three)   # hon
last_three = language[3:]
print(last_three)   # hon
```

#### –†–∞–∑–≤–æ—Ä–æ—Ç —Å—Ç—Ä–æ–∫–∏

–í Python –º—ã –º–æ–∂–µ–º —Å –ª–µ–≥–∫–æ—Å—Ç—å—é —Ä–∞–∑–≤–µ—Ä–Ω—É—Ç—å —Å—Ç—Ä–æ–∫—É!

```py
greeting = 'Hello, World!'
```

#### –ü—Ä–æ–ø—É—Å–∫ —Å–∏–º–≤–æ–ª–æ–≤ –≤ —Å—Ä–µ–∑–∞—Ö

–ú—ã  –º–æ–∂–µ–º –ø—Ä–æ–ø—É—Å–∫–∞—Ç—å –Ω–µ–∫–æ—Ç–æ—Ä—ã–µ —Å–∏–º–≤–æ–ª—ã –≤ —Å—Ä–µ–∑–∞—Ö –±–ª–∞–≥–æ–¥–∞—Ä—è —à–∞–≥—É —Å—Ä–µ–∑–∞.

```py
language = 'Python'
pto = language[0:6:2] # –±–µ—Ä–µ–º 0, 2 –∏ 4 –∏–Ω–¥–µ–∫—Å—ã

```

### –ú–µ—Ç–æ–¥—ã —Å—Ç—Ä–æ–∫

–í Python —Å—É—â–µ—Å—Ç–≤—É–µ—Ç –º–Ω–æ–∂–µ—Å—Ç–≤–æ –º–µ—Ç–æ–¥–æ–∫ —Å—Ç—Ä–æ–∫, –¥–∞–≤–∞–π—Ç–µ —Ä–∞—Å—Å–º–æ—Ç—Ä–∏–º –Ω–µ–∫–æ—Ç–æ—Ä—ã–µ –∏–∑ –Ω–∏—Ö:

- capitalize(): –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç –ø–µ—Ä–≤—ã–π —Å–∏–º–≤–æ–ª —Å—Ç—Ä–æ–∫–∏ –≤ –∑–∞–≥–ª–∞–≤–Ω—É—é –±—É–∫–≤—É.

```py
challenge = 'python –∑–∞ 30 –¥–Ω–µ–π'
print(challenge.capitalize()) # 'Python –∑–∞ 30 –¥–Ω–µ–π'
```

- count(): –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –≤—Ö–æ–∂–¥–µ–Ω–∏–π –ø–æ–¥—Å—Ç—Ä–æ–∫–∏ –≤ —Å—Ç—Ä–æ–∫—É. –ú–µ—Ç–æ–¥ –∏–º–µ–µ—Ç —Å–ª–µ–¥—É—é—â–∏–π —Å–∏–Ω—Ç–∞–∫—Å–∏—Å: `count(substring, start=..., end=...)`. –ü–∞—Ä–∞–º–µ—Ç—Ä `start`` –ø—Ä–∏–Ω–∏–º–∞–µ—Ç –Ω–∞—á–∞–ª—å–Ω—ã–π –∏–Ω–¥–µ–∫—Å –¥–ª—è –ø–æ–∏—Å–∫–∞, –∞ –ø–∞—Ä–∞–º–µ—Ç—Ä `end` –ø—Ä–∏–Ω–∏–º–∞–µ—Ç –ø–æ—Å–ª–µ–¥–Ω–∏–π –∏–Ω–¥–µ–∫—Å, –≥–¥–µ –±—É–¥–µ—Ç –≤–µ—Å—Ç–∏—Å—å –ø–æ–∏—Å–∫ –ø–æ–¥—Å—Ç—Ä–æ–∫–∏.

```py
challenge = 'Python –∑–∞ 30 –¥–Ω–µ–π'
print(challenge.count('P')) # 1
print(challenge.count('–∞', 7, 14)) # 1
print(challenge.count('—ë')) # 0
```

- endswith(): –ü—Ä–æ–≤–µ—Ä—è–µ—Ç, –∑–∞–∫–∞–Ω—á–∏–≤–∞–µ—Ç—Å—è –ª–∏ —Å—Ç—Ä–æ–∫–∞ —É–∫–∞–∑–∞–Ω–Ω—ã–º –Ω–∞–±–æ—Ä–æ–º —Å–∏–º–≤–æ–ª–æ–≤

```py
challenge = 'Python –∑–∞ 30 –¥–Ω–µ–π'
print(challenge.endswith('–µ–π'))   # True
print(challenge.endswith('!')) # False
```

- expandtabs(): –ó–∞–º–µ–Ω—è–µ—Ç —Å–∏–º–≤–æ–ª —Ç–∞–±—É–ª—è—Ü–∏–∏ –Ω–∞ —É–∫–∞–∑–∞–Ω–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø—Ä–æ–±–µ–ª–æ–≤

```py
challenge = 'Python\t–∑–∞\t30\t–¥–Ω–µ–∏ÃÜ'
print(challenge)   # Python	–∑–∞	30	–¥–Ω–µ–∏ÃÜ
print(challenge.expandtabs(1))  # Python –∑–∞ 30 –¥–Ω–µ–∏ÃÜ
```

- find(): –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∏–Ω–¥–µ–∫—Å –ø–µ—Ä–≤–æ–≥–æ –≤—Ö–æ–∂–¥–µ–Ω–∏—è –ø–æ–¥—Å—Ç—Ä–æ–∫–∏, –µ—Å–ª–∏ –ø–æ–¥—Å—Ç—Ä–æ–∫–∞ –Ω–µ –Ω–∞–π–¥–µ–Ω–∞, –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç -1.

```py
challenge = 'Python –∑–∞ 30 –¥–Ω–µ–π'
print(challenge.find('y'))  # 1
print(challenge.find('th')) # 2
```

- rfind(): –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∏–Ω–¥–µ–∫—Å –ø–æ—Å–ª–µ–¥–Ω–µ–≥–æ –≤—Ö–æ–∂–¥–µ–Ω–∏—è –ø–æ–¥—Å—Ç—Ä–æ–∫–∏, –µ—Å–ª–∏ –ø–æ–¥—Å—Ç—Ä–æ–∫–∞ –Ω–µ –Ω–∞–π–¥–µ–Ω–∞, –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç -1.

```py
challenge = 'Python –∑–∞ 30 –¥–Ω–µ–π'

print(challenge.rfind('–µ'))  # 15

```

- format(): —Ñ–æ—Ä–º–∞—Ç–∏—Ä—É–µ—Ç —Å—Ç—Ä–æ–∫—É. –ü–æ–¥—Ä–æ–±–Ω–µ–µ –æ —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏–∏ —Å—Ç—Ä–æ–∫ –º–æ–∂–Ω–æ —É–∑–Ω–∞—Ç—å –ø–æ [—Å—Å—ã–ª–∫–µ](https://teletype.in/@pythontalk/f_strings).

```py
first_name = '–ü–∏—Ç–µ—Ä'
last_name = '–ü–∞—Ä–∫–µ—Ä'
age = 250
job = '–ß–µ–ª–æ–≤–µ–∫-–ø–∞—É–∫'
country = '–†–æ—Å—Å–∏—è'
sentence = '–ú–µ–Ω—è –∑–æ–≤—É—Ç {} {}. –ú–Ω–µ {} –ª–µ—Ç. –Ø {}. –ú–æ—è –ª—é–±–∏–º–∞—è —Å—Ç—Ä–∞–Ω–∞ {}.'.format(first_name, last_name, age, job, country)
print(sentence) # –ú–µ–Ω—è –∑–æ–≤—É—Ç –ü–∏—Ç–µ—Ä –ü–∞—Ä–∫–µ—Ä. –ú–Ω–µ 250 –ª–µ—Ç. –Ø –ß–µ–ª–æ–≤–µ–∫-–ø–∞—É–∫. –ú–æ—è –ª—é–±–∏–º–∞—è —Å—Ç—Ä–∞–Ω–∞ –†–æ—Å—Å–∏—è.

radius = 10
pi = 3.14
```
